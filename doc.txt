//安装conda环境
wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh  
bash Miniconda3-latest-Linux-x86_64.sh -u
source ~/bashrc      
source ~/.bashrc
conda create -n test python=3.8    
//安装所需的包
conda activate test
conda install pytorch==2.0.0 torchvision==0.15.0 torchaudio==2.0.0 pytorch-cuda=11.8 -c pytorch -c nvidia # CPU Only
pip install -r requirements.txt
//如果提示哪个包还没有安装则需要补上

//在浏览器中运行
python webApp.py 


//在后端运行
pip3 install gunicorn 
nohub gunicorn --workers 1 --bind 0.0.0.0:5000 webApp:app
